Working on more JavaScript exercises. Quick example solution for this problem. This doesn't feel like the most elegant solution, so I will try to streamline it.

Return true if the string in the first element of the array contains all of the letters of the string in the second element of the array.

function mutation(arr) {
  var test = arr[1].toLowerCase();
  var mutate = arr[0].toLowerCase();
  for (var i = 0; i < test.length; i++) {
    if (mutate.indexOf(test[i]) !== -1) {
      continue;
    } else return false;
  }
  return true;
}
